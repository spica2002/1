```mermaid
    %% 사용자 및 인터페이스
    User[<color:white><b>사용자 (공정 엔지니어/데이터 과학자)</b></color>] :::user
    AA[<color:white><b>AI 에이전트 (LLM 기반)</b>\n지능형 질의 해석 및 도구 오케스트레이션</color>] :::agent

    %% MCP 도구 아키텍처 경계
    subgraph MCP_Ecosystem [나노팹 공정 분석 MCP 도구 생태계]
        direction TB

        %% 1. 공정데이터 질의 MCP
        subgraph Server_Data [1. 공정데이터 질의 MCP 서버]
            direction TB
            T_History[<b>공정 이력 추적 도구</b>\n- 웨이퍼/실험 단위 흐름 추적\n- 공정 간 연계 데이터 조회] :::tool
            T_Param[<b>공정 파라미터 조건 질의 도구</b>\n- 수치 범위 질의 (Temp, Press, Time...)\n- 다중 조건 복합 질의] :::tool
            T_Process[<b>단위 공정별 데이터 질의 도구</b>\n- 이온주입, 증착, 포토, 식각, 세정\n- 장비/날짜/조건별 조회\n- 레코드/메타데이터 반환] :::tool
            T_Stats[<b>요약 통계 자동 생성 도구</b>\n- 질의 결과 통계 (건수, 분포, 기초통계량)] :::tool
        end

        %% 2. 분석 모델 서비스 MCP
        subgraph Server_Model [2. 분석 모델 서비스 MCP 서버]
            direction TB
            subgraph Pred_Models [공정 예측/추천 AI 모델]
                T_Ion_Pred[이온주입 예측] :::tool
                T_Dep_Pred[증착 예측] :::tool
                T_Photo_Pred[포토 예측] :::tool
                T_Etch_Pred[식각 예측] :::tool
            end
            T_Anomaly[<b>시계열 이상탐지 AI 도구</b>\n- 센서 데이터 이상탐지\n- Golden Recipe 대비 편차 반환] :::tool
            T_Defect[<b>웨이퍼 결함 분류 AI 도구</b>\n- 이미지 기반 결함 분류\n- 분류 결과 및 신뢰도 반환] :::tool
            
            ML_Mgmt[<font color=black>모델 버전 관리 & 추론 이력 로깅</font>] :::support
            Standard_IO[<font color=black>입출력 데이터 형식 표준화</font>] :::support
        end

        %% 3. 시각화 및 차트생성 MCP
        subgraph Server_Viz [3. 시각화 및 차트생성 MCP 서버]
            direction TB
            T_ChartGen[<b>데이터 시각화 차트 자동 생성 도구</b>\n- Hist, Box, Scatter, Contour, Heatmap\n- 트렌드 차트 (시계열)\n- Parallel Coordinates Plot] :::tool
            T_WaferMap[<b>웨이퍼맵 시각화 도구</b>\n- 결함/측정값 분포] :::tool
            
            Func_Rec[<font color=black>최적 차트 유형 자동 추천</font>] :::support
            Func_Render[<font color=black>차트 이미지/인터랙티브 렌더링</font>] :::support
            Func_Insight[<font color=black>캡션 및 인사이트 자동 생성</font>] :::support
        end

    end

    %% 외부 데이터 및 인프라
    subgraph Data_Infra [나노팹 데이터 인프라]
        DB_Process[(공정 DB\nRaw Data)] :::db
        DB_MES[(MES/이력 DB)] :::db
        F_Image[[결함 이미지\n스토리지]] :::db
        Model_Repo[[AI 모델\n레포지토리]] :::db
    end

    %% 연결선 (흐름)
    User ==>|자연어 질의 (예: A웨이퍼 식각 이상 분석해줘)| AA
    
    %% Agent <-> MCP Data
    AA <==>|MCP Protocol\n(json-rpc)| Server_Data
    T_History -.->|SQL/Query| DB_MES
    T_Process -.->|SQL/Query| DB_Process
    T_Param -.->|SQL/Query| DB_Process
    
    %% Agent <-> MCP Model
    AA <==>|MCP Protocol\n(json-rpc)| Server_Model
    Server_Model -.->|모델 로드/추론| Model_Repo
    T_Defect -.->|이미지 로드| F_Image
    
    %% Agent <-> MCP Viz
    AA <==>|MCP Protocol\n(json-rpc)| Server_Viz

    %% 내부 지원 기능 연결
    Server_Model -.-> ML_Mgmt
    Server_Model -.-> Standard_IO
    Server_Viz -.-> Func_Rec
    Server_Viz -.-> Func_Render
    Server_Viz -.-> Func_Insight

    %% 최종 결과 반환
    AA ==>|분석 리포트 (통계+예측+차트)| User

    %% 스타일 정의
    classDef user fill:#333,stroke:#333,stroke-width:2px,rx:10,ry:10;
    classDef agent fill:#1a237e,stroke:#1a237e,stroke-width:2px,rx:10,ry:10;
    classDef tool fill:#f1f8e9,stroke:#558b2f,stroke-width:1px,rx:5,ry:5,color:#333,font-size:12px;
    classDef support fill:#f5f5f5,stroke:#9e9e9e,stroke-width:1px,stroke-dasharray: 5 5,color:#333,font-size:11px;
    classDef db fill:#fff3e0,stroke:#ef6c00,stroke-width:1px,stroke-dasharray: 5 5,color:#333,font-size:12px;
    
    linkStyle 0,16 ==> stroke:#01579b,stroke-width:3px;
    linkStyle 1,4,13 <=> stroke:#d32f2f,stroke-width:2px,stroke-dasharray: 5 5;
